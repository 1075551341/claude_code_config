### Task 1.5: scripts/sync.ps1 — 全量软链接同步重写 [P0]

**文件**: `C:\Users\DELL\.claude\scripts\sync.ps1`

**操作**: 重写脚本，从 L0-only 扩展到全量 rules/skills/agents/CLAUDE.md 同步

**核心逻辑**:
```powershell
# sync.ps1 v2.0 — 全量软链接同步到多编辑器
param([switch]$DryRun, [switch]$ForceCopy)

$source = "$env:USERPROFILE\.claude"
$targets = @{
    "cursor"   = "$env:USERPROFILE\.cursor"
    "devin"    = "$env:USERPROFILE\.claude\.devin"
    "qoder"    = "$env:USERPROFILE\.qoder"
    "trae"     = "$env:USERPROFILE\.trae"
    "codearts" = "$env:USERPROFILE\.config\codeartsdoer"
}

$syncItems = @(
    @{src="rules";       dst="rules";       ext=".md"},
    @{src="skills";      dst="skills";      ext=""},    # 整个目录
    @{src="agents";      dst="agents";      ext=".md"},
    @{src="CLAUDE.md";   dst="CLAUDE.md";   ext=""}     # 单文件
)

function Sync-Directory {
    param($srcDir, $dstDir, $ext, $editorName)
    # 1. 目标存在 → 删除同类型同名文件（去重）
    # 2. 优先 New-Item -Type SymbolicLink（需开发者模式）
    # 3. 失败 → Copy-Item 硬拷贝
    # 4. --DryRun → 仅报告，不执行
}

# 主流程
foreach ($editor in $targets.Keys) {
    foreach ($item in $syncItems) {
        Sync-Directory -srcDir "$source\$($item.src)" `
                       -dstDir "$($targets[$editor])\$($item.dst)" `
                       -ext $item.ext `
                       -editorName $editor
    }
}
```
